1. Get the DeepWiki URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "deepwiki.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status by running:
  kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "deepwiki.fullname" . }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "deepwiki.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "deepwiki.fullname" . }} 8090:{{ .Values.service.port }}
  echo "Visit http://127.0.0.1:8090"
{{- end }}

2. Check the health of the deployment:
  kubectl --namespace {{ .Release.Namespace }} get pods -l "app.kubernetes.io/name={{ include "deepwiki.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

3. View logs:
  kubectl --namespace {{ .Release.Namespace }} logs -f deployment/{{ include "deepwiki.fullname" . }}

4. Test health endpoint:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "deepwiki.fullname" . }} 8090:{{ .Values.service.port }} &
  curl http://localhost:8090/health

5. Test descriptor endpoint:
  curl http://localhost:8090/descriptor
