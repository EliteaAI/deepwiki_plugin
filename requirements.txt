# --- Unified DB (sqlite-vec) ------------------------------------
# sqlite-vec provides vector KNN search as a loadable SQLite extension.
# pysqlite3 ships its own libsqlite3 built with ENABLE_LOAD_EXTENSION,
# which the stock macOS / official-site Python lacks.  On Linux (Docker)
# the stdlib sqlite3 already supports extensions, but pysqlite3 is
# harmless there (compiles fine, gives a newer SQLite 3.51 vs 3.40).
# unified_db.py imports: try pysqlite3 as sqlite3 / except: import sqlite3
#
# Build note: pysqlite3 compiles from source — requires libsqlite3-dev
# in Dockerfile.worker builder stage (already added).
#
# NOTE: sqlite-vec 0.1.6 aarch64 wheel is broken (ships 32-bit ARM .so).
# On Apple Silicon Docker (linux/aarch64) vec will fail to load.
# Works fine on x86_64 (production) and macOS native (via pysqlite3).
sqlite-vec==0.1.9
pysqlite3==0.6.0
